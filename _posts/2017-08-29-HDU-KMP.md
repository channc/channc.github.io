[HDU 1711](http://acm.hdu.edu.cn/showproblem.php?pid=1711)

裸KMP，附上一个模板

```
#include <cstdio>
using namespace std;
const int maxn = 1000000+6, maxm = 10000+6;
int nxt[maxm], p[maxm], s[maxn], m, n, t;
void getnext() {
	nxt[0] = -1;
	int plen = m, k = -1, j = 0;
	while (j < plen) {
		if (k == -1 || p[j] == p[k]) { k++, j++; nxt[j] = k; }
		else k = nxt[k];
	}
}
int kmp() {
	getnext();
	int slen = n, plen = m;
	int i = 0, j = 0;
	while (i < slen) {
		if (j == -1 || s[i] == p[j]) ++i, ++j;
		else j = nxt[j];
		if (j == plen) return i-j;
	}
	return -1;
}
int main() {
	scanf("%d", &t); while (t--) {
		scanf("%d%d", &n, &m);
		for (int i = 0; i < n; ++i) scanf("%d", &s[i]);
		for (int i = 0; i < m; ++i) scanf("%d", &p[i]);
		int ans = kmp();
		if (ans == -1) printf("-1\n");
		else printf("%d\n", ans+1);
	}
	return 0;
}
```

[HDU 1711](http://acm.hdu.edu.cn/showproblem.php?pid=1711)

KMP变形，所做的更改即在判断最后是否匹配完成时记录并将 `j` 置0。

```
const int maxn = 1000000+6, maxm = 10000+6;
int nxt[maxm], m, n, t;
char p[maxm], s[maxn];
void getnext() {
	nxt[0] = -1;
	int plen = m, k = -1, j = 0;
	while (j < plen) {
		if (k == -1 || p[j] == p[k]) { k++, j++; nxt[j] = k; }
		else k = nxt[k];
	}
}
int kmp() {
	getnext();
	int slen = n, plen = m, cnt = 0;
	int i = 0, j = 0;
	while (i < slen) {
		if (j == -1 || s[i] == p[j]) ++i, ++j;
		else j = nxt[j];
		// if (j == plen) return i-j;
		**if (j == plen) { cnt++; j = 0; }**
	}
	//return -1;
	return cnt;
}
int main() {
	while (EOF != scanf("%s", s)) {
		if (s[0] == '#') return 0;
		scanf("%s", p);
		n = strlen(s), m = strlen(p);
		int ans = kmp();
		printf("%d\n", ans);
	}
	return 0;
}
```
