**next数组的一个重要性质** 若`i%(i-next[i])==0`则`i`前面的串为循环子串，且`i-next[i]`表示字符串中第i个位置的循环节长度。

```
const int maxm = 1000000+5;
int nxt[maxm], m, n, t, result;
char p[maxm], s[maxn];
void getnext() {
	nxt[0] = -1;
	int plen = m, k = -1, j = 0;
	while (j < plen) {
		if (k == -1 || p[j] == p[k]) { k++, j++; nxt[j] = k; }
		else k = nxt[k];
	}
}
int main() {
	int kase = 1;
	while (scanf("%d", &t)!=EOF && t) {
		scanf("%s", p);
		m = t;
		getnext();
		printf("Test case #%d\n", kase++);
		for (int i = 1; i <= m; ++i) {
			int t = i-nxt[i];
			if (i%t == 0 && i/t > 1)
				printf("%d %d\n", i, i/t);
		}
		printf("\n");
	}
	return 0;
}
```
